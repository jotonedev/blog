{{- /* Image render hook. */ -}} {{- $src := .Destination -}} {{- $alt := .PlainText | default .Text -}} {{- $isRemote := or (hasPrefix $src "http://") (hasPrefix $src "https://") (hasPrefix $src "//") -}} {{- $img := false -}} {{- if not $isRemote -}} {{- $img = .Page.Resources.GetMatch $src -}} {{- if not $img }}{{ $img = resources.Get (strings.TrimPrefix "/" $src) }}{{ end -}} {{- if and $img (ne $img.ResourceType "image") }}{{ $img = false }}{{ end -}} {{- end -}} {{- with .Title }}{{ end -}}
