{{- /* Renders the photo feed, newest first. Images come from photos/ (mounted at assets/photos), metadata from data/photos.yaml. The container is 880px, so 1760 covers a 2x display; anything wider than the source is skipped rather than upscaled. */ -}} {{- $widths := slice 480 880 1320 1760 -}} {{- $sizes := "(max-width: 928px) calc(100vw - 3rem), 880px" -}} {{- $items := slice -}} {{- range hugo.Data.photos -}} {{- if not .file -}} {{- warnf "photo_list: entry in data/photos.yaml has no `file` key, skipping" -}} {{- else -}} {{- $img := resources.Get (printf "photos/%s" .file) -}} {{- if not $img -}} {{- warnf "photo_list: %q is listed in data/photos.yaml but missing from photos/" .file -}} {{- else if ne $img.ResourceType "image" -}} {{- warnf "photo_list: %q is not an image" .file -}} {{- else -}} {{- /* Capture date: YAML wins, EXIF fills in. Photos with no EXIF report a zero time, which is truthy in a template, so test it explicitly. */ -}} {{- $date := "" -}} {{- with .date }}{{ $date = time.AsTime . }}{{ end -}} {{- if not $date -}} {{- with $img.Meta }}{{ if not .Date.IsZero }}{{ $date = .Date }}{{ end }}{{ end -}} {{- end -}} {{- if not $date -}} {{- warnf "photo_list: %q has no `date` in data/photos.yaml and none in its EXIF, so it sorts last" .file -}} {{- end -}} {{- /* Camera: YAML wins, else EXIF make + model. Models often already carry the make ("Canon EOS R6"), so only prepend it when they do not. */ -}} {{- $device := .device | default "" -}} {{- if not $device -}} {{- $make := "" -}} {{- $model := "" -}} {{- with $img.Meta -}} {{- with .Exif.Make }}{{ $make = title (lower (trim (string .) " ")) }}{{ end -}} {{- with .Exif.Model }}{{ $model = trim (string .)
